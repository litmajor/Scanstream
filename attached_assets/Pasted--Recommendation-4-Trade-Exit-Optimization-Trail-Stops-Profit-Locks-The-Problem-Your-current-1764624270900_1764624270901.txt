The Problem:
Your current exits are probably:

Fixed take-profit: +2%
Fixed stop-loss: -5%
No adjustment as trade progresses

This means:

You exit winners too early (BTC runs 10%, you got 2%)
You hold losers too long (-5% stop when -2% would save you)

The Solution: Dynamic Trail Stops
pythonclass IntelligentExitManager:
    
    def __init__(self, entry_price, initial_stop_pct, atr):
        self.entry_price = entry_price
        self.highest_price = entry_price
        self.current_stop = entry_price * (1 - initial_stop_pct)
        self.atr = atr
        self.profit_locked = 0
        
    def update(self, current_price, time_held_hours):
        """
        Called every candle/update
        Dynamically adjusts stop based on price action
        """
        
        # Track highest price
        if current_price > self.highest_price:
            self.highest_price = current_price
        
        current_profit_pct = ((current_price - self.entry_price) / self.entry_price) * 100
        
        # STAGE 1: Initial Risk (0-1% profit)
        if current_profit_pct < 1.0:
            # Keep initial stop, don't trail yet
            pass
        
        # STAGE 2: Breakeven Protection (1-2% profit)
        elif current_profit_pct < 2.0:
            # Move stop to breakeven
            new_stop = self.entry_price
            self.current_stop = max(self.current_stop, new_stop)
            self.profit_locked = 0
        
        # STAGE 3: Profit Lock (2-4% profit)
        elif current_profit_pct < 4.0:
            # Lock in 50% of profit
            profit_to_lock = (self.highest_price - self.entry_price) * 0.5
            new_stop = self.entry_price + profit_to_lock
            self.current_stop = max(self.current_stop, new_stop)
            self.profit_locked = 50
        
        # STAGE 4: Aggressive Trail (4%+ profit)
        else:
            # Trail stop at 1.5 Ã— ATR below highest price
            trail_distance = self.atr * 1.5
            new_stop = self.highest_price - trail_distance
            self.current_stop = max(self.current_stop, new_stop)
            self.profit_locked = ((new_stop - self.entry_price) / self.entry_price) * 100
        
        # TIME-BASED EXIT
        # If trade held >7 days and profit <3%, consider exit
        if time_held_hours > 168 and current_profit_pct < 3.0:
            return {
                'action': 'EXIT',
                'reason': 'Trade exhausted - time limit reached',
                'profit_pct': current_profit_pct
            }
        
        # CHECK IF STOP HIT
        if current_price <= self.current_stop:
            return {
                'action': 'EXIT',
                'reason': 'Stop loss hit',
                'profit_pct': current_profit_pct
            }
        
        return {
            'action': 'HOLD',
            'current_stop': self.current_stop,
            'profit_locked_pct': self.profit_locked
        }
