python# CONCEPT: Historical Move Analysis Per Asset Per Timeframe

class AssetVelocityProfile:
    def __init__(self, asset):
        self.asset = asset
        self.profiles = self.calculate_historical_moves()
    
    def calculate_historical_moves(self):
        # Analyze last 2 years of data
        data = get_historical_data(self.asset, period='2y')
        
        timeframes = {
            '1D': 1,
            '3D': 3,
            '7D': 7,
            '14D': 14,
            '21D': 21,
            '30D': 30
        }
        
        profiles = {}
        
        for name, days in timeframes.items():
            moves = []
            
            # Calculate rolling moves over each window
            for i in range(len(data) - days):
                start_price = data[i]['close']
                end_price = data[i + days]['close']
                
                dollar_move = abs(end_price - start_price)
                pct_move = (dollar_move / start_price) * 100
                
                moves.append({
                    'dollar': dollar_move,
                    'percent': pct_move,
                    'direction': 'up' if end_price > start_price else 'down'
                })
            
            # Statistical analysis
            profiles[name] = {
                'avg_dollar_move': np.mean([m['dollar'] for m in moves]),
                'median_dollar_move': np.median([m['dollar'] for m in moves]),
                'avg_pct_move': np.mean([m['percent'] for m in moves]),
                'median_pct_move': np.median([m['percent'] for m in moves]),
                'p25': np.percentile([m['dollar'] for m in moves], 25),
                'p75': np.percentile([m['dollar'] for m in moves], 75),
                'p90': np.percentile([m['dollar'] for m in moves], 90),
                'max_move': max([m['dollar'] for m in moves]),
                'up_days': sum(1 for m in moves if m['direction'] == 'up') / len(moves)
            }
        
        return profiles

1. Set Realistic Profit Targets Based on Holding Period
pythondef set_profit_target(asset, trade_type, entry_price):
    velocity = AssetVelocityProfile(asset)
    
    if trade_type == TradeType.SCALP:
        # 1-day move, take 50-75th percentile
        expected_move = velocity.profiles['1D']['p75']
        target = entry_price + (expected_move * 0.5)  # Take half the p75 move
        
    elif trade_type == TradeType.DAY:
        # 1-day move, take 75th percentile
        expected_move = velocity.profiles['1D']['p75']
        target = entry_price + (expected_move * 0.8)
        
    elif trade_type == TradeType.SWING:
        # 7-day move, take 75th percentile
        expected_move = velocity.profiles['7D']['p75']
        target = entry_price + expected_move
        
    elif trade_type == TradeType.POSITION:
        # 21-day move, aim for 90th percentile
        expected_move = velocity.profiles['21D']['p90']
        target = entry_price + expected_move
    
    return target

2. Dynamic Exit Based on Move Completion
pythondef check_exit_signal(asset, entry_price, current_price, days_held, trade_type):
    velocity = AssetVelocityProfile(asset)
    
    # Calculate actual move so far
    actual_move = current_price - entry_price
    
    # Get expected move for time window
    if days_held <= 1:
        expected = velocity.profiles['1D']['avg_dollar_move']
    elif days_held <= 3:
        expected = velocity.profiles['3D']['avg_dollar_move']
    elif days_held <= 7:
        expected = velocity.profiles['7D']['avg_dollar_move']
    elif days_held <= 14:
        expected = velocity.profiles['14D']['avg_dollar_move']
    else:
        expected = velocity.profiles['30D']['avg_dollar_move']
    
    # Calculate move completion %
    completion_pct = (actual_move / expected) * 100
    
    # EXIT LOGIC:
    
    # If you've captured 80%+ of expected move, consider exiting
    if completion_pct >= 80:
        return {
            'action': 'EXIT',
            'reason': f'Captured {completion_pct}% of expected {days_held}D move',
            'profit': actual_move
        }
    
    # If held 2x target window and only 30% of move, cut losses
    expected_days = get_expected_days(trade_type)
    if days_held > expected_days * 2 and completion_pct < 30:
        return {
            'action': 'EXIT',
            'reason': 'Trade exhausted, move not materializing',
            'profit': actual_move
        }
    
    # If showing momentum exhaustion (volume drop + RSI divergence)
    if detect_exhaustion(asset) and completion_pct > 50:
        return {
            'action': 'EXIT',
            'reason': 'Momentum exhaustion, lock in gains',
            'profit': actual_move
        }
    
    return {'action': 'HOLD'}
