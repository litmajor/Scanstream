class AdaptiveMLEnsemble:
    def __init__(self):
        # Train separate models for each market regime
        self.trending_model = train_model(regime='trending')
        self.choppy_model = train_model(regime='choppy')
        self.volatile_model = train_model(regime='volatile')
        
        # Meta-model to select which model to use
        self.regime_classifier = train_regime_classifier()
    
    def predict(self, asset, features):
        # Classify current market regime
        regime = self.regime_classifier.predict(features)
        
        # Use appropriate model
        if regime == 'trending':
            prediction = self.trending_model.predict(features)
            confidence_boost = 1.15  # Boost confidence in trending markets
        elif regime == 'choppy':
            prediction = self.choppy_model.predict(features)
            confidence_boost = 0.85  # Reduce confidence in choppy markets
        else:  # volatile
            prediction = self.volatile_model.predict(features)
            confidence_boost = 0.95
        
        return prediction * confidence_boost
