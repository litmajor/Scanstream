        """

        if trade_result.profit > 0:
            self.wins += 1
            xp_earned = self.calculate_xp(trade_result)
            self.xp += xp_earned
            self.total_profit += trade_result.profit

            self.confidence = min(1.0, self.confidence + 0.05)
            
        else:
            self.losses += 1

        is_breakout = momentum_data['price'] > momentum_data['resistance']
        volume_spike = momentum_data['volume'] > momentum_data['avg_volume'] * 2
        
        # Skill-based enhancements
        if 'pattern_recognition' in self.skills:

                # Reduce confidence in non-trending regimes
                pattern_quality *= 0.7

        if 'multi_timeframe_confirmation' in self.abilities:
            mtf_score = self.check_multiple_timeframes(momentum_data)
            pattern_quality *= mtf_score

                'action': 'BUY',

                stats['win_rate'] * 30 +

            self.leaderboard[agent_name]['points'] = score
            
            # Assign rank

🎯 TIER 5: CROSS-AGENT COOPERATION
Agent Council - Collective Intelligence
pythonclass AgentCouncil:
    """
    Facilitates cooperation between agents
    Vote on trades, share intelligence, coordinate strategies
    """
    
    def __init__(self, agents):
        self.agents = agents
        self.voting_threshold = 0.6  # 60% consensus needed
    
    def vote_on_trade(self, asset, trade_signal):
        """
        All agents vote on whether to take a trade
        """
        votes = {
            'buy': 0,
            'sell': 0,
            'skip': 0
        }
        
        vote_weights = {}
        
        for agent in self.agents:
            # Each agent evaluates the trade from their perspective
            agent_opinion = agent.evaluate_trade(asset, trade_signal)
            
            # Weight vote by agent's skill level and confidence
            weight = agent.level * agent.confidence
            vote_weights[agent.name] = weight
            
            if agent_opinion == 'BUY':
                votes['buy'] += weight
            elif agent_opinion == 'SELL':
                votes['sell'] += weight
            else:
                votes['skip'] += weight
        
        # Calculate consensus
        total_weight = sum(vote_weights.values())
        buy_consensus = votes['buy'] / total_weight
        sell_consensus = votes['sell'] / total_weight
        
        # Decision
        if buy_consensus > self.voting_threshold:
            return {
                'decision': 'BUY',
                'consensus': buy_consensus,
                'supporting_agents': [a.name for a in self.agents if a.evaluate_trade(asset, trade_signal) == 'BUY']
            }
        elif sell_consensus > self.voting_threshold:
            return {
                'decision': 'SELL',
                'consensus': sell_consensus,
                'supporting_agents': [a.name for a in self.agents if a.evaluate_trade(asset, trade_signal) == 'SELL']
            }
        else:
            return {
                'decision': 'SKIP',
                'consensus': 0,
                'reason': 'No consensus reached'
            }
    
    def share_intelligence(self):
        """
        Agents share what they've learned
        """
        for agent in self.agents:
            # Agent broadcasts what it knows
            intelligence = {
                'agent': agent.name,
                'best_patterns': agent.get_best_patterns(),
                'market_insights': agent.get_market_insights(),
                'risk_warnings': agent.get_risk_warnings()
            }
            
            # Other agents can learn from this
            for other_agent in self.agents:
                if other_agent != agent:
                    other_agent.receive_intelligence(intelligence)

        active_combos = []

            required_agents = combo_data['agents']
            present_agents = [a for a in agents if a.name in required_agents]

                })
