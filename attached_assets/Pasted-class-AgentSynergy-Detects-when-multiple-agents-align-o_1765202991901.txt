class AgentSynergy:
    """
    Detects when multiple agents align on a setup
    and applies synergy bonuses
    """
    
    COMBOS = {
        'triple_bottom_hunter': {
            'agents': ['SUPPORT_SNIPER', 'REVERSAL_MASTER', 'ML_ORACLE'],
            'condition': 'all_agree_buy_at_support',
            'min_combined_level': 30,
            'bonuses': {
                'win_rate': +0.15,
                'position_size': 1.3,  # 30% larger positions
                'confidence': +0.10
            }
        },
        'tsunami': {
            'agents': ['BREAKOUT_HUNTER', 'TREND_RIDER', 'ML_ORACLE'],
            'condition': 'breakout_with_strong_trend',
            'min_combined_level': 30,
            'bonuses': {
                'profit_target': 1.5,  # 50% larger targets
                'trailing_stop': True,
                'confidence': +0.20
            }
        },
        'the_trap': {
            'agents': ['BREAKOUT_HUNTER', 'REVERSAL_MASTER', 'TREND_RIDER'],
            'condition': 'failed_breakout_detected',
            'min_combined_level': 45,
            'bonuses': {
                'inverse_trade': True,  # Fade the breakout
                'confidence': +0.25,
                'position_size': 1.5
            }
        }
    }
    
    def check_for_combos(self, agents, signal):
        """
        Check if multiple agents form a synergy combo
        """
        active_combos = []
        
        for combo_name, combo_data in self.COMBOS.items():
            # Check if required agents are present
            required_agents = combo_data['agents']
            present_agents = [a for a in agents if a.name in required_agents]
            
            if len(present_agents) < len(required_agents):
                continue
            
            # Check level requirement
            combined_level = sum(a.level for a in present_agents)
            if combined_level < combo_data['min_combined_level']:
                continue
            
            # Check condition
            if self.check_condition(combo_data['condition'], present_agents, signal):
                active_combos.append({
                    'name': combo_name,
                    'agents': present_agents,
                    'bonuses': combo_data['bonuses']
                })
        
        return active_combos

# EXAMPLE IN ACTION:

agents = [BREAKOUT_HUNTER, TREND_RIDER, ML_ORACLE]
signal = {
    'asset': 'BTC/USDT',
    'pattern': 'breakout',
    'regime': 'BULL_TRENDING',
    'trend_strength': 'STRONG'
}

combos = synergy_system.check_for_combos(agents, signal)

if combos:
    print(f"🎊 COMBO ACTIVATED: {combos[0]['name']}")
    print(f"Participating agents: {[a.name for a in combos[0]['agents']]}")
    print(f"Bonuses applied:")
    for bonus_type, bonus_value in combos[0]['bonuses'].items():
        print(f"  - {bonus_type}: {bonus_value}")
    
    # Apply bonuses to trade
    signal['confidence'] *= (1 + combos[0]['bonuses']['confidence'])
    signal['profit_target'] *= combos[0]['bonuses']['profit_target']
