──────────────────────────────────────────────────────────────────────────

const exitDecision = exitMaster.analyzeExit({
  entryPrice: 42500,
  currentPrice: 43100,
  atr: 500,
  signalType: 'BUY',
  profitPercent: 0.014,
  timeHeldHours: 3,
  microstructure: marketData,
});

if (exitDecision.action === 'EXIT') {
  console.log(`Exit at $${exitDecision.exitPrice}`);
  console.log(`Reason: ${exitDecision.reason}`);
}


USE CASE 2: Opposition-Based Exit
──────────────────────────────────────────────────────────────────────────

const opposition = oppositionReader.analyzeOpposition({
  currentPrice: 43100,
  supportLevels: [42500, 41800],
  resistanceLevels: [44200, 45000],
  volume: 12000,
  priceVelocity: 0.005,
  volatility: 0.015,
  timeToSupport: 8,
});

if (opposition.supportStrength < 0.3) {
  console.log('Support weak - exit before breakdown');
}


USE CASE 3: Consensus Exit (Highest Confidence)
